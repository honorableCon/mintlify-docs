# Kiriku — Setup Self-Hosted KYC Stack

> **Stack** : MiniFASNet (liveness) + DeepFace/ArcFace (face match) + yente/OpenSanctions (AML)  
> **Prérequis** : Docker, Node.js 20+, Python 3.10+, NestJS existant

---

## Architecture cible

```
kiriku-api (NestJS)
    │
    ├── POST /v1/liveness/check   → Python microservice (MiniFASNet)  :8001
    ├── POST /v1/match/face       → Python microservice (DeepFace)    :8002
    ├── POST /v1/aml/screen       → yente container (OpenSanctions)   :8080
    └── POST /v1/kyc/score        → Orchestration interne NestJS
```

Les 3 outils tournent en microservices Python appelés via HTTP depuis NestJS. Simple, découplé, scalable.

```python
from fastapi import FastAPI, UploadFile, File, HTTPException
from fastapi.responses import JSONResponse
import cv2
import numpy as np
import onnxruntime as ort
from PIL import Image
import io
import time

app = FastAPI(title="Kiriku Liveness Service")

# Charger les modèles MiniFASNet au démarrage
try:
    session_small = ort.InferenceSession("models/MiniFASNetV2.onnx")
    session_large = ort.InferenceSession("models/MiniFASNetV2SE.onnx")
    print("✅ Modèles MiniFASNet chargés")
except Exception as e:
    print(f"❌ Erreur chargement modèles: {e}")
    session_small = session_large = None

def preprocess_image(image_bytes: bytes) -> np.ndarray:
    """Prétraitement image pour MiniFASNet"""
    pil_img = Image.open(io.BytesIO(image_bytes)).convert("RGB")
    img = np.array(pil_img)
    
    # Resize à 80x80 (taille attendue MiniFASNet)
    img = cv2.resize(img, (80, 80))
    img = img.astype(np.float32) / 255.0
    
    # Normalisation ImageNet
    mean = np.array([0.485, 0.456, 0.406])
    std = np.array([0.229, 0.224, 0.225])
    img = (img - mean) / std
    
    # HWC → CHW → NCHW
    img = img.transpose(2, 0, 1)
    img = np.expand_dims(img, axis=0)
    return img.astype(np.float32)

def predict_liveness(image_bytes: bytes) -> dict:
    """Prédiction liveness via les deux modèles (ensemble)"""
    if not session_small or not session_large:
        raise HTTPException(500, "Modèles non chargés")
    
    img = preprocess_image(image_bytes)
    
    # Inférence modèle small
    input_name = session_small.get_inputs()[0].name
    pred_small = session_small.run(None, {input_name: img})[0]
    score_small = float(pred_small[0][1])  # probabilité "real"
    
    # Inférence modèle large
    input_name = session_large.get_inputs()[0].name
    pred_large = session_large.run(None, {input_name: img})[0]
    score_large = float(pred_large[0][1])
    
    # Ensemble : moyenne pondérée
    final_score = (score_small * 0.4) + (score_large * 0.6)
    
    is_live = final_score > 0.6
    attack_type = None if is_live else "spoof_attack"
    
    return {
        "isLive": bool(is_live),
        "confidence": round(final_score, 4),
        "attackType": attack_type,
        "scoreSmall": round(score_small, 4),
        "scoreLarge": round(score_large, 4),
    }

@app.post("/liveness/check")
async def check_liveness(file: UploadFile = File(...)):
    start = time.time()
    
    if not file.content_type.startswith("image/"):
        raise HTTPException(400, "Fichier image requis")
    
    image_bytes = await file.read()
    
    if len(image_bytes) > 10 * 1024 * 1024:
        raise HTTPException(400, "Image > 10MB")
    
    result = predict_liveness(image_bytes)
    result["processingMs"] = round((time.time() - start) * 1000)
    
    return JSONResponse(result)

@app.get("/health")
def health():
    return {
        "status": "ok",
        "modelsLoaded": session_small is not None and session_large is not None
    }
```

```python
from fastapi import FastAPI, HTTPException
from fastapi.responses import JSONResponse
from pydantic import BaseModel
from deepface import DeepFace
import base64
import io
import time
import numpy as np
from PIL import Image
import tempfile
import os

app = FastAPI(title="Kiriku Face Match Service")

# Pré-charger le modèle ArcFace au démarrage
print("⏳ Chargement modèle ArcFace...")
try:
    # Force le téléchargement du modèle au démarrage
    dummy = np.zeros((112, 112, 3), dtype=np.uint8)
    tmp = tempfile.NamedTemporaryFile(suffix=".jpg", delete=False)
    Image.fromarray(dummy).save(tmp.name)
    try:
        DeepFace.represent(tmp.name, model_name="ArcFace", enforce_detection=False)
    except:
        pass
    os.unlink(tmp.name)
    print("✅ Modèle ArcFace chargé")
except Exception as e:
    print(f"⚠️  Pré-chargement: {e}")

class FaceMatchRequest(BaseModel):
    referenceImage: str   # base64
    liveImage: str        # base64

def base64_to_tempfile(b64: str) -> str:
    """Décoder base64 → fichier temporaire"""
    # Supprimer le header data:image/...;base64, si présent
    if "," in b64:
        b64 = b64.split(",")[1]
    
    img_bytes = base64.b64decode(b64)
    tmp = tempfile.NamedTemporaryFile(suffix=".jpg", delete=False)
    tmp.write(img_bytes)
    tmp.close()
    return tmp.name

@app.post("/face/match")
async def match_faces(req: FaceMatchRequest):
    start = time.time()
    
    ref_path = live_path = None
    try:
        ref_path = base64_to_tempfile(req.referenceImage)
        live_path = base64_to_tempfile(req.liveImage)
        
        result = DeepFace.verify(
            img1_path=ref_path,
            img2_path=live_path,
            model_name="ArcFace",
            distance_metric="cosine",
            enforce_detection=True,   # True en prod — rejette si pas de visage détecté
            detector_backend="retinaface",
        )
        
        # Convertir la distance cosine en score de similarité 0-1
        distance = result["distance"]
        threshold = result["threshold"]
        similarity = max(0.0, 1.0 - (distance / threshold))
        
        confidence = "high" if similarity > 0.85 else "medium" if similarity > 0.70 else "low"
        
        return JSONResponse({
            "match": bool(result["verified"]),
            "similarity": round(float(similarity), 4),
            "confidence": confidence,
            "distance": round(float(distance), 4),
            "threshold": round(float(threshold), 4),
            "processingMs": round((time.time() - start) * 1000),
        })
    
    except ValueError as e:
        # Pas de visage détecté
        raise HTTPException(422, f"Visage non détecté: {str(e)}")
    except Exception as e:
        raise HTTPException(500, f"Erreur analyse: {str(e)}")
    finally:
        for path in [ref_path, live_path]:
            if path and os.path.exists(path):
                os.unlink(path)

@app.post("/face/extract-embedding")
async def extract_embedding(req: dict):
    """Extraire l'embedding pour le Face Search 1:N"""
    start = time.time()
    tmp_path = None
    try:
        tmp_path = base64_to_tempfile(req["image"])
        embedding_obj = DeepFace.represent(
            img_path=tmp_path,
            model_name="ArcFace",
            enforce_detection=True,
            detector_backend="retinaface",
        )
        return JSONResponse({
            "embedding": embedding_obj[0]["embedding"],  # vecteur 512 dimensions
            "processingMs": round((time.time() - start) * 1000),
        })
    except Exception as e:
        raise HTTPException(500, str(e))
    finally:
        if tmp_path and os.path.exists(tmp_path):
            os.unlink(tmp_path)

@app.get("/health")
def health():
    return {"status": "ok", "model": "ArcFace"}
```

```typescript
import { Injectable, Logger, BadRequestException } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';

export interface LivenessResult {
  isLive: boolean;
  confidence: number;
  attackType: string | null;
  processingMs: number;
}

@Injectable()
export class LivenessService {
  private readonly logger = new Logger(LivenessService.name);
  private readonly baseUrl: string;

  constructor(private readonly configService: ConfigService) {
    this.baseUrl = this.configService.get<string>(
      'LIVENESS_SERVICE_URL',
      'http://localhost:8001',
    );
  }

  async check(imageBuffer: Buffer, mimeType: string): Promise<LivenessResult> {
    const formData = new FormData();
    const blob = new Blob([imageBuffer], { type: mimeType });
    formData.append('file', blob, 'image.jpg');

    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 10_000);

    try {
      const response = await fetch(`${this.baseUrl}/liveness/check`, {
        method: 'POST',
        body: formData,
        signal: controller.signal,
      });

      if (!response.ok) {
        const err = await response.json().catch(() => ({}));
        throw new BadRequestException(`Liveness service error: ${JSON.stringify(err)}`);
      }

      return response.json() as Promise<LivenessResult>;
    } catch (error) {
      if (error.name === 'AbortError') {
        throw new BadRequestException('Liveness service timeout');
      }
      throw error;
    } finally {
      clearTimeout(timeout);
    }
  }

  async isServiceHealthy(): Promise<boolean> {
    try {
      const res = await fetch(`${this.baseUrl}/health`, { signal: AbortSignal.timeout(3000) });
      const data = await res.json();
      return data.status === 'ok' && data.modelsLoaded;
    } catch {
      return false;
    }
  }
}
```

```typescript
import {
  Controller, Post, UseGuards, UseInterceptors,
  UploadedFile, BadRequestException, Request,
} from '@nestjs/common';
import { FileInterceptor } from '@nestjs/platform-express';
import { LivenessService } from './liveness.service';
import { ApiKeyOrJwtGuard } from '../auth/guards/api-key-or-jwt.guard';
import { RateLimitGuard } from '../auth/guards/rate-limit.guard';

@Controller('v1/liveness')
@UseGuards(ApiKeyOrJwtGuard, RateLimitGuard)
export class LivenessController {
  constructor(private readonly livenessService: LivenessService) {}

  @Post('check')
  @UseInterceptors(FileInterceptor('file'))
  async check(
    @UploadedFile() file: Express.Multer.File,
    @Request() req: any,
  ) {
    if (!file) throw new BadRequestException('Fichier image requis');

    const validTypes = ['image/jpeg', 'image/png', 'image/webp'];
    if (!validTypes.includes(file.mimetype)) {
      throw new BadRequestException('Format non supporté (jpeg, png, webp)');
    }

    return this.livenessService.check(file.buffer, file.mimetype);
  }
}
```

```typescript
import { Injectable, Logger, BadRequestException } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';

export interface FaceMatchResult {
  match: boolean;
  similarity: number;
  confidence: 'low' | 'medium' | 'high';
  distance: number;
  processingMs: number;
}

@Injectable()
export class FaceMatchService {
  private readonly logger = new Logger(FaceMatchService.name);
  private readonly baseUrl: string;

  constructor(private readonly configService: ConfigService) {
    this.baseUrl = this.configService.get<string>(
      'FACE_MATCH_SERVICE_URL',
      'http://localhost:8002',
    );
  }

  async compare(
    referenceImage: Buffer | string,   // buffer ou base64
    liveImage: Buffer | string,
  ): Promise<FaceMatchResult> {
    const toBase64 = (img: Buffer | string) =>
      Buffer.isBuffer(img) ? img.toString('base64') : img;

    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 15_000);

    try {
      const response = await fetch(`${this.baseUrl}/face/match`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          referenceImage: toBase64(referenceImage),
          liveImage: toBase64(liveImage),
        }),
        signal: controller.signal,
      });

      if (!response.ok) {
        const err = await response.json().catch(() => ({}));
        throw new BadRequestException(`Face match error: ${JSON.stringify(err)}`);
      }

      return response.json() as Promise<FaceMatchResult>;
    } catch (error) {
      if (error.name === 'AbortError') {
        throw new BadRequestException('Face match service timeout');
      }
      throw error;
    } finally {
      clearTimeout(timeout);
    }
  }

  async extractEmbedding(imageBase64: string): Promise<number[]> {
    const response = await fetch(`${this.baseUrl}/face/extract-embedding`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ image: imageBase64 }),
      signal: AbortSignal.timeout(15_000),
    });

    if (!response.ok) throw new BadRequestException('Embedding extraction failed');
    const data = await response.json();
    return data.embedding;
  }
}
```

**`src/modules/aml/aml.service.ts`** :
```typescript
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { InjectModel } from '@nestjs/mongoose';
import { Model, Types } from 'mongoose';
import { AmlCheck, AmlCheckDocument } from './schemas/aml-check.schema';

export interface AmlScreenRequest {
  firstName: string;
  lastName: string;
  dateOfBirth?: string;
  nationality?: string;
  documentNumber?: string;
}

export interface AmlHit {
  entityId: string;
  name: string;
  score: number;
  datasets: string[];
  topics: string[];
}

export interface AmlResult {
  screened: boolean;
  riskLevel: 'low' | 'medium' | 'high' | 'critical';
  hits: AmlHit[];
  listsChecked: number;
  checkedAt: string;
}

@Injectable()
export class AmlService {
  private readonly logger = new Logger(AmlService.name);
  private readonly yenteUrl: string;

  constructor(
    private readonly configService: ConfigService,
    @InjectModel(AmlCheck.name)
    private readonly amlCheckModel: Model<AmlCheckDocument>,
  ) {
    this.yenteUrl = this.configService.get<string>(
      'AML_SERVICE_URL',
      'http://localhost:8080',
    );
  }

  async screen(
    userId: string,
    req: AmlScreenRequest,
  ): Promise<AmlResult> {
    const query = `${req.firstName} ${req.lastName}`;

    const response = await fetch(
      `${this.yenteUrl}/match/entities?q=${encodeURIComponent(query)}&limit=5&threshold=0.7`,
      { signal: AbortSignal.timeout(10_000) },
    );

    if (!response.ok) {
      throw new Error(`AML service error: ${response.status}`);
    }

    const data = await response.json();

    const hits: AmlHit[] = (data.results || []).map((r: any) => ({
      entityId: r.id,
      name: r.caption,
      score: r.score,
      datasets: r.datasets || [],
      topics: r.properties?.topics || [],
    }));

    // Filtrer par score > 0.80 (réduire faux positifs)
    const relevantHits = hits.filter(h => h.score > 0.80);

    const riskLevel = this.calculateRiskLevel(relevantHits);

    const result: AmlResult = {
      screened: true,
      riskLevel,
      hits: relevantHits,
      listsChecked: data.total || 0,
      checkedAt: new Date().toISOString(),
    };

    // Sauvegarder en DB pour audit
    await this.amlCheckModel.create({
      userId: new Types.ObjectId(userId),
      query: { firstName: req.firstName, lastName: req.lastName },
      result,
    });

    return result;
  }

  private calculateRiskLevel(hits: AmlHit[]): 'low' | 'medium' | 'high' | 'critical' {
    if (hits.length === 0) return 'low';

    const maxScore = Math.max(...hits.map(h => h.score));
    const hasSanctions = hits.some(h =>
      h.datasets.some(d => ['us_ofac_sdn', 'un_sc_sanctions', 'eu_fsf'].includes(d))
    );

    if (hasSanctions && maxScore > 0.95) return 'critical';
    if (hasSanctions) return 'high';
    if (maxScore > 0.90) return 'medium';
    return 'low';
  }
}
```

### 4.4 KYC Score — orchestrateur

**`src/modules/kyc/kyc.service.ts`** :
```typescript
import { Injectable, Logger } from '@nestjs/common';
import { LivenessService } from '../liveness/liveness.service';
import { FaceMatchService } from '../face-match/face-match.service';
import { AmlService } from '../aml/aml.service';
import { ExtractionsService } from '../extractions/extractions.service';

export interface KycScoreRequest {
  extractionId: string;
  selfieBase64: string;
  enableAml?: boolean;
}

export interface KycScoreResult {
  kycScore: number;
  riskLevel: 'low' | 'medium' | 'high' | 'critical';
  recommendation: 'APPROVE' | 'REVIEW' | 'REJECT';
  breakdown: {
    ocrConfidence: number;
    livenessScore: number | null;
    faceMatchScore: number | null;
    amlClear: boolean | null;
  };
  verifiedAt: string;
}

@Injectable()
export class KycService {
  private readonly logger = new Logger(KycService.name);

  constructor(
    private readonly livenessService: LivenessService,
    private readonly faceMatchService: FaceMatchService,
    private readonly amlService: AmlService,
  ) {}

  async score(userId: string, req: KycScoreRequest): Promise<KycScoreResult> {
    const results = await Promise.allSettled([
      // Liveness sur le selfie
      this.livenessService.check(
        Buffer.from(req.selfieBase64, 'base64'),
        'image/jpeg',
      ),
      // AML si demandé (on passe null sinon)
      req.enableAml ? this.amlService.screen(userId, {
        firstName: '', // à récupérer depuis l'extraction
        lastName: '',
      }) : Promise.resolve(null),
    ]);

    const livenessResult = results[0].status === 'fulfilled' ? results[0].value : null;
    const amlResult = results[1].status === 'fulfilled' ? results[1].value : null;

    // Face match : toujours tenté
    let faceMatchResult = null;
    try {
      // Le selfie vs photo stockée dans l'extraction
      faceMatchResult = await this.faceMatchService.compare(
        req.selfieBase64,  // selfie
        req.selfieBase64,  // photo du doc — à récupérer depuis extraction
      );
    } catch (e) {
      this.logger.warn(`Face match failed: ${e.message}`);
    }

    // Calculer le score
    const ocrConfidence = 0.95; // depuis l'extraction
    const livenessScore = livenessResult?.confidence ?? null;
    const faceMatchScore = faceMatchResult?.similarity ?? null;
    const amlClear = amlResult ? amlResult.riskLevel === 'low' : null;

    const kycScore = this.calculateScore({
      ocrConfidence,
      livenessScore,
      faceMatchScore,
      amlClear,
    });

    return {
      kycScore,
      riskLevel: kycScore >= 80 ? 'low' : kycScore >= 60 ? 'medium' : 'high',
      recommendation: kycScore >= 75 ? 'APPROVE' : kycScore >= 50 ? 'REVIEW' : 'REJECT',
      breakdown: { ocrConfidence, livenessScore, faceMatchScore, amlClear },
      verifiedAt: new Date().toISOString(),
    };
  }

  private calculateScore(breakdown: {
    ocrConfidence: number;
    livenessScore: number | null;
    faceMatchScore: number | null;
    amlClear: boolean | null;
  }): number {
    let score = 0;
    let maxScore = 0;

    // OCR (15%)
    score += breakdown.ocrConfidence * 15;
    maxScore += 15;

    // Liveness (25%)
    if (breakdown.livenessScore !== null) {
      score += breakdown.livenessScore * 25;
      maxScore += 25;
    }

    // Face match (25%)
    if (breakdown.faceMatchScore !== null) {
      score += breakdown.faceMatchScore * 25;
      maxScore += 25;
    }

    // AML (20%)
    if (breakdown.amlClear !== null) {
      score += (breakdown.amlClear ? 1 : 0) * 20;
      maxScore += 20;
    }

    // Normaliser sur 100
    return maxScore > 0 ? Math.round((score / maxScore) * 100) : 0;
  }
}
```
